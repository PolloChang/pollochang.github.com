Arch Linux 安裝

2024-08-02 Arch linux

環境

  • KVM
    • CPU: 2 core
    • RAM: 2G

環境調查事項

  • 開機方式: bios_grub or UEFI

環境規劃

硬碟

  • swap: 4G
  • bios_grub: 1M
  • /boot: 1GB

下載

wget https://linux.cs.nctu.edu.tw/archlinux/iso/2024.08.01/archlinux-2024.08.01-x86_64.iso

安裝 SOP

# 檢查網路
ping -c 3 8.8.8.8
# 檢查是否為UEFI模式開機
cat /sys/firmware/efi/fw_platform_size
# 分割硬碟
fdisk -l
# Create a GPT partition table:
parted /dev/vda mklabel gpt
parted /dev/vda

mkpart primary 1MiB 2MiB
set 1 bios_grub on
mkpart primary 2MiB 100%
# ctrl + D

# 檢查結果
fdisk -l

mkfs.ext4 /dev/sd(磁碟編號)(分割區編號)

# SWAP
mkswap /dev/sda1
swapon /dev/sda1


mount /dev/vda /mnt

# 初始化根目錄
pacstrap /mnt base base-devel

# Wi-Fi 硬體
pacstrap /mnt iw dialog wpa_supplicant wpa_actiond

genfstab -U /mnt > /mnt/etc/fstab
# 設定時間
timedatectl set-ntp true
# 安裝 linux 穩定版
sudo pacman -S linux-lts
# 安裝 Intel 的微碼
sudo pacman -S intel-ucode
# 安裝 AMD 的微碼
sudo pacman -S linux-firmware
# 安裝基本套件
pacman -S vim grub openssh firewalld neofetch networkmnager
# 設定時區
ln -sf /usr/share/zoneinfo/Asia/Taiwan /etc/localtime
# 將系統時間設定為 UTC
hwclock -w --utc
# 語系設定
/etc/locale.gen
locale-gen
# 主機名稱
echo "想要的主機名稱" >/etc/hostname

## 主機帳號
useradd


systemctl start sshd
systemctl enable sshd
systemctl enable firewalld
systemctl enable NetworkManager
sudo firewall-cmd --permanent --zone=public --change-interface=enp1s0 &&\
firewall-cmd --reload

exit

reboot

# 配置系统默認编辑器: /etc/profile
export EDITOR='vim'

桌面安裝

pacman -S plasma-meta konsole dolphin
systemctl enable sddm
